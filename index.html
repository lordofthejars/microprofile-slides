JAX-RS Example

@Path("/")
public class GreetingResource {

  @Inject @Informal Greeting greeting;
  
  @GET
  @Path("greeting/{name}")
  @Produces(MediaType.TEXT_PLAIN)
  public Response getMessageByUser(@PathParam("name") String name) {
    return Response.ok(greeting.greet(name)).build();
  }
}
